```
    <script type="text/javascript">
        function sort (arr) {
            for (var i = 0;i<arr.length;i++) {
                for (var j = 0; j < arr.length-i-1; j++) {
                    if (arr[j]>arr[j+1]) {
                        var temp=arr[j];
                        arr[j]=arr[j+1];
                        arr[j+1]=temp;
                    }
                }
            }
            return arr;
        }
        var arr=[2,5,3,1,7,8,78,89];
        sort(arr);
    
    </script>
```
