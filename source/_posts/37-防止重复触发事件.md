```
// 防重复点击
stopReapeatEvent=function (obj) {
    if (obj.callFlag) return true;
    obj.callFlag=!obj.callFlag;
    clearTimeout(obj.iTime);
    obj.iTime=setTimeout(function(){
        obj.callFlag=!obj.callFlag;
    }, 1000);
}
```

```
//调用
if(stopReapeatEvent(arguments.callee))  return ;
```
