		function fn() {
			var _arg=[].slice.call(arguments);
			return function() {
				if(arguments.length==0){
					return _arg.reduce((a,b)=>a+b);
				}else{
					return fn.apply(null,_arg.concat([].slice.call(arguments)));
				}
			}
		}
